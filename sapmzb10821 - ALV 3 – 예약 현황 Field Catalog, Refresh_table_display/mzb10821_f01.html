*&---------------------------------------------------------------------*
*&      Form  GET_VALUE_TEXT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*      -->P_LT_DCLASS  text
*      -->P_LS_DCLASS  text
*      -->P_0446   text
*      -->P_0447   text
*      <--P_GS_DATA  text
*----------------------------------------------------------------------*
FORM get_value_text USING pt_domain TYPE tt_domain
                              VALUE(p_value)
                              VALUE(p_value_t)
                     CHANGING ps_domain TYPE dd07v
                              cs_data LIKE gs_data.

  CLEAR: ps_domain.

  FIELD-SYMBOLS <fs>.

  DATA lv_field TYPE string.

  CONCATENATE 'CS_DATA-' p_value INTO lv_field.
  ASSIGN (lv_field) TO <fs>.

  READ TABLE pt_domain INTO ps_domain
    WITH KEY domvalue_l = <fs>.

  IF sy-subrc <> 0.
    RETURN.
  ENDIF.

  CLEAR lv_field.
  UNASSIGN <fs>.
  CONCATENATE 'CS_DATA-' p_value_t INTO lv_field.
  ASSIGN (lv_field) TO <fs>.

  <fs> = ps_domain-ddtext.

ENDFORM.
