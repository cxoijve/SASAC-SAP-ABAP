    ENDIF.

  IF zsb003301-customid IS NOT INITIAL.
    SELECT a~carrid a~connid a~fldate a~bookid a~customid
           a~custtype a~class a~loccuram a~loccurkey a~order_date
           b~price b~currency
      FROM sbook AS a
      INNER JOIN sflight AS b
      ON a~carrid = b~carrid
      AND a~connid = b~connid
      AND a~fldate = b~fldate
      INTO CORRESPONDING FIELDS OF TABLE gt_book
     WHERE a~carrid = zsb003301-carrid
       AND a~connid EQ zsb003301-connid
       AND a~fldate BETWEEN zsb003301-fldate_b AND zsb003301-fldate_e
       AND a~customid = zsb003301-customid.
  ELSE.
    SELECT a~carrid a~connid a~fldate a~bookid a~customid
         a~custtype a~class a~loccuram a~loccurkey a~order_date
         b~price b~currency
    FROM sbook AS a
    INNER JOIN sflight AS b
    ON a~carrid = b~carrid
    AND a~connid = b~connid
    AND a~fldate = b~fldate
    INTO CORRESPONDING FIELDS OF TABLE gt_book
