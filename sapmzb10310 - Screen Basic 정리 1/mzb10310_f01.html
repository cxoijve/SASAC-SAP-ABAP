FORM get_airline_name .
* DATA: GT_SCARR TYPE TABLE OF SCARR,
*      GS_SCARR TYPE SCARR.

  SELECT SINGLE carrid carrname
    FROM scarr
    INTO CORRESPONDING FIELDS OF zsb101301
    WHERE carrid = zsb101301-carrid.
  IF sy-subrc <> 0.
    MESSAGE s000(zmbc10) DISPLAY LIKE 'E'.
    RETURN.
  ENDIF.

ENDFORM.
