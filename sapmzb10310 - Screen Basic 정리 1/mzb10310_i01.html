MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN 'CANC'.
      SET SCREEN 0.
      LEAVE SCREEN.

    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'SEARCH'.
      CLEAR zsb101302.

      SELECT SINGLE a~carrid carrname connid airpfrom airpto
        FROM spfli AS a INNER JOIN scarr AS b
        ON a~carrid = b~carrid
        INTO CORRESPONDING FIELDS OF zsb101302
        WHERE a~carrid = zsb101301-carrid
          AND connid = zsb101301-connid.

      SELECT SINGLE c~name AS airpfrom_t
                    d~name AS airpto_t
        FROM spfli AS a
        INNER JOIN sairport AS c ON a~airpfrom = c~id
        INNER JOIN sairport AS d ON a~airpto = d~id
        INTO (zsb101302-airpfrom_t, zsb101302-airpto_t)
        WHERE a~carrid = zsb101302-carrid
                AND connid = zsb101302-connid.

*      READ TABLE gt_scarr INTO gs_scarr WITH KEY carrid = zsb101301-carrid.

      CALL SCREEN 0200.


    WHEN 'ENTER'.


      PERFORM get_airline_name.
*
    WHEN OTHERS.

  ENDCASE.
ENDMODULE.
