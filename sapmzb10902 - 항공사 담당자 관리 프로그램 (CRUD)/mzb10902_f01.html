FORM get_spfli .
  SELECT a~carrid a~connid a~airpfrom a~airpto
         b~name AS airpfrom_t c~name AS airpto_t
    FROM spfli AS a

    INTO CORRESPONDING FIELDS OF TABLE gt_spfli.
