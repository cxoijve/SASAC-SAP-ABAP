FORM get_airline_name .
  " Get Airline Name
  SELECT SINGLE carrid carrname
    FROM scarr
    INTO CORRESPONDING FIELDS OF zsb101303
    WHERE carrid = zsb101303-carrid.

  IF sy-subrc <> 0.
    MESSAGE s000(zmbc10) DISPLAY LIKE 'E'.
  ENDIF.
ENDFORM.

FORM set_default .
  zsb101303-carrid = 'KA'.
  zsb101303-connid = 0017.

  SELECT carrid connid fldate
    FROM sflight
    INTO CORRESPONDING FIELDS OF TABLE gt_sflight
    WHERE carrid = zsb101303-carrid
    AND connid = zsb101303-connid.

  SORT gt_sflight BY carrid connid fldate DESCENDING.

  READ TABLE gt_sflight INTO gs_sflight
  WITH KEY carrid = zsb101303-carrid
           connid = zsb101303-connid.

  IF sy-subrc = 0.
    zsb101303-fldate = gs_sflight-fldate.
  ENDIF.


ENDFORM.

FORM get_flight_info .
  CLEAR zsb101304.

  SELECT SINGLE a~carrid carrname connid fldate price currency
    FROM sflight AS a INNER JOIN scarr AS b
    ON a~carrid = b~carrid
    INTO CORRESPONDING FIELDS OF zsb101304
    WHERE a~carrid = zsb101303-carrid
    AND connid = zsb101303-connid
    AND fldate = zsb101303-fldate.


  IF sy-subrc <> 0.
    MESSAGE s000(zmbc10) DISPLAY LIKE 'E'.
  ENDIF.
ENDFORM.

FORM get_connection_info .
  zsb101305-carrid = zsb101304-carrid.
  zsb101305-carrname = zsb101304-carrname.

  SELECT SINGLE airpfrom airpto
    FROM spfli
    INTO (zsb101305-airpfrom, zsb101305-airpto)
    WHERE carrid = zsb101304-carrid
    AND connid = zsb101304-connid.

  DATA: lt_sairport TYPE TABLE OF sairport
         WITH HEADER LINE.

  SELECT id name
    FROM sairport
    INTO CORRESPONDING FIELDS OF TABLE lt_sairport.

  READ TABLE lt_sairport[] WITH KEY id = zsb101305-airpfrom
  INTO lt_sairport.

  zsb101305-airpfrom_t = lt_sairport-name.

  READ TABLE lt_sairport WITH KEY id = zsb101305-airpto.

  zsb101305-airpto_t = lt_sairport-name.

ENDFORM.
