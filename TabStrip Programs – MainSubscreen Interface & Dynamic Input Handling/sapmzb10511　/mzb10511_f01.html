FORM get_emp_name .
  CLEAR: zsb102011-name, zsb102011-lname, zsb102011-fname.

  SELECT SINGLE lname fname
    FROM ztb0001
    INTO CORRESPONDING FIELDS OF zsb102011
    WHERE pernr = zsb102011-pernr.


  " 사원명 취득
***  CONCATENATE zsb102011-lname zsb102011-fname
***  INTO zsb102011-name.
**  zsb102011-name = zsb102011-lname && zsb102011-fname.
  PERFORM concate_emp_name USING zsb102011-lname zsb102011-fname
                           CHANGING zsb102011-name.
ENDFORM.

FORM get_dep_info .
  CLEAR zsb102013.

  SELECT SINGLE depid phone dflag
    FROM ztb0002
    INTO CORRESPONDING FIELDS OF zsb102013
    WHERE depid = zsb102012-orgeh.

  IF sy-subrc <> 0.

  ENDIF.

FORM concate_emp_name USING VALUE(pa_lname)
                            VALUE(pa_fname)
                      CHANGING cv_name .
  CLEAR cv_name.

  CONCATENATE pa_lname pa_fname
  INTO cv_name.

ENDFORM.

  CLEAR:   zsb102012-gesch_m, zsb102012-gesch_f.
